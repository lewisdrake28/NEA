@page
@model IndexModel
@using Microsoft.AspNetCore.Http;
@{
    ViewData["Title"] = "Home page";
}

<html>

<body>

    @{
        //after form is submitted
        if (HttpMethods.IsPost(Request.Method))
        {
            //getting numerical values
            string input1 = Request.Form["num1"];
            string input2 = Request.Form["num2"];
            int num1 = int.Parse(input1);
            int num2 = int.Parse(input2);

            //getting submit buttons values
            //if a button is pressed the value will be the name
            //e.g when + is pressed, isAdd = "+"
            var isAdd = Request.Form["add"];
            var isSubtract = Request.Form["subtract"];

            //dealing with each button press
            if (isAdd == " + ")
            {
                var ans = num1 + num2;
                <p>@ans</p>
            }
            else if (isSubtract == " - ")
            {
                var ans = num1 - num2;
                <p>@ans</p>
            }
        }
        //before form is submitted
        else
        {
            //building the form
            <form method="post" action="">
                <p>Number 1:</p>
                <input type="number" value="" name="num1" /><br>
                <p>Number 2:</p>
                <input type="number" value="" name="num2" /><br><br>
                @*submit buttons*@
                <input type="submit" value=" + " name="add" />
                <input type="submit" value=" - " name="subtract" />
            </form>
        }
    }

</body>

</html>
